107 static inline double CoinCpuTime()
108 {
109 #ifdef COIN_DOING_DIFFS
110  // when trying to see differences between runs it can be helpful
111  return 0.0;
112 #endif
113  double cpu_temp;
114 #if defined(_MSC_VER) || defined(__MSVCRT__)
115 #ifdef HAVE_SDK
116  FILETIME creation;
117  FILETIME exit;
118  FILETIME kernel;
119  FILETIME user;
120  GetProcessTimes(GetCurrentProcess(), &creation, &exit, &kernel, &user);
121  double t = user.dwHighDateTime * TWO_TO_THE_THIRTYTWO + user.dwLowDateTime;
122  return t / 10000000.0;
123 #else
124  unsigned int ticksnow; /* clock_t is same as int */
125  ticksnow = (unsigned int)clock();
126  cpu_temp = (double)((double)ticksnow / CLOCKS_PER_SEC);
127 #endif
128 
129 #else
130  struct rusage usage;
131 #ifdef ZEROFAULT
132  usage.ru_utime.tv_sec = 0;
133  usage.ru_utime.tv_usec = 0;
134 #endif
135  getrusage(RUSAGE_SELF, &usage);
136  cpu_temp = static_cast< double >(usage.ru_utime.tv_sec);
137  cpu_temp += 1.0e-6 * (static_cast< double >(usage.ru_utime.tv_usec));
138 #endif
139  return cpu_temp;
140 }
